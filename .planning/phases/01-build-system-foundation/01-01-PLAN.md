  <action>
Create CMakeLists.txt in the project root. The file must:

1. **Minimum CMake version and project declaration:**
   ```cmake
   cmake_minimum_required(VERSION 3.22)
   project(PluginTemplate VERSION 1.0.0)
   ```

2. **Add JUCE subdirectory** (from submodule):
   ```cmake
   add_subdirectory(libs/JUCE)
   ```

3. **Add clap-juce-extensions subdirectory** (from submodule):
   ```cmake
   add_subdirectory(libs/clap-juce-extensions)
   ```

4. **Define the plugin with juce_add_plugin**, specifying all three formats. Use general audio effect defaults as required:
   ```cmake
   juce_add_plugin(PluginTemplate
       COMPANY_NAME "YourCompany"
       PLUGIN_MANUFACTURER_CODE Mfrc
       PLUGIN_CODE Plug
       FORMATS VST3 Standalone
       IS_SYNTH FALSE
       IS_MIDI_EFFECT FALSE
       NEEDS_MIDI_INPUT FALSE
       NEEDS_MIDI_OUTPUT FALSE
       EDITOR_WANTS_KEYBOARD_FOCUS FALSE
       COPY_PLUGIN_AFTER_BUILD FALSE
       VST3_CATEGORIES "Fx"
       PLUGIN_NAME "Plugin Template"
   )
   ```

   Note: CLAP format is added via clap-juce-extensions after the initial juce_add_plugin call (see step 6), not in the FORMATS list.

5. **Add placeholder source files** (stub files that will be replaced in Phase 2, but must exist for the build to succeed). Create minimal stub sources inline using cmake's file(WRITE) or reference them:
   ```cmake
   target_sources(PluginTemplate PRIVATE
       src/PluginProcessor.cpp
       src/PluginProcessor.h
       src/PluginEditor.cpp
       src/PluginEditor.h
   )
   ```
   Also create the actual stub source files (see Task 3 below — actually, create them in this task as part of making the build compile):

   - `src/PluginProcessor.h` — minimal juce::AudioProcessor subclass declaration
   - `src/PluginProcessor.cpp` — minimal implementation returning juce::AudioProcessor::BusesProperties() with stereo in/out
   - `src/PluginEditor.h` — minimal juce::AudioProcessorEditor subclass declaration
   - `src/PluginEditor.cpp` — minimal implementation

   The stubs just need to compile — Phase 2 will flesh them out.

6. **Wire up JUCE module dependencies:**
   ```cmake
   target_compile_definitions(PluginTemplate
       PUBLIC
       JUCE_WEB_BROWSER=0
       JUCE_USE_CURL=0
       JUCE_VST3_CAN_REPLACE_VST2=0
   )

   target_link_libraries(PluginTemplate
       PRIVATE
       juce::juce_audio_utils

       juce::juce_recommended_lto_flags
       juce::juce_recommended_warning_flags
   )
   ```

7. **Add CLAP support via clap-juce-extensions:**
   ```cmake
   clap_juce_extensions_plugin(
       TARGET PluginTemplate
       CLAP_ID "com.yourcompany.plugintemplate"
       CLAP_FEATURES audio-effect
       CLAP_MANUAL_URL "https://yourcompany.com"
       CLAP_SUPPORT_URL "https://yourcompany.com"
   )
   ```

The file must NOT contain IS_SYNTH=TRUE or IS_MIDI_EFFECT=TRUE — the defaults are FALSE, matching BUILD-04.

Also create the minimal stub source files in src/ so the build does not fail on missing files:

**src/PluginProcessor.h:**
```cpp
#pragma once
#include <juce_audio_processors/juce_audio_processors.h>

class PluginProcessor : public juce::AudioProcessor
{
public:
    PluginProcessor();
    ~PluginProcessor() override;

    void prepareToPlay(double sampleRate, int samplesPerBlock) override;
    void releaseResources() override;
    void processBlock(juce::AudioBuffer<float>&, juce::MidiBuffer&) override;

    juce::AudioProcessorEditor* createEditor() override;
    bool hasEditor() const override { return true; }

    const juce::String getName() const override { return JucePlugin_Name; }
    bool acceptsMidi() const override { return false; }
    bool producesMidi() const override { return false; }
    double getTailLengthSeconds() const override { return 0.0; }

    int getNumPrograms() override { return 1; }
    int getCurrentProgram() override { return 0; }
    void setCurrentProgram(int) override {}
    const juce::String getProgramName(int) override { return {}; }
    void changeProgramName(int, const juce::String&) override {}

    void getStateInformation(juce::MemoryBlock& destData) override;
    void setStateInformation(const void* data, int sizeInBytes) override;

private:
    JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(PluginProcessor)
};
```

**src/PluginProcessor.cpp:**
```cpp
#include "PluginProcessor.h"
#include "PluginEditor.h"

PluginProcessor::PluginProcessor()
    : AudioProcessor(BusesProperties()
          .withInput("Input", juce::AudioChannelSet::stereo(), true)
          .withOutput("Output", juce::AudioChannelSet::stereo(), true))
{
}

PluginProcessor::~PluginProcessor() {}

void PluginProcessor::prepareToPlay(double, int) {}
void PluginProcessor::releaseResources() {}

void PluginProcessor::processBlock(juce::AudioBuffer<float>& buffer, juce::MidiBuffer&)
{
    juce::ScopedNoDenormals noDenormals;
    // Pass-through: copy input to output unchanged
    for (int ch = 0; ch < buffer.getNumChannels(); ++ch)
        buffer.copyFrom(ch, 0, buffer.getReadPointer(ch), buffer.getNumSamples());
}

juce::AudioProcessorEditor* PluginProcessor::createEditor()
{
    return new PluginEditor(*this);
}

void PluginProcessor::getStateInformation(juce::MemoryBlock&) {}
void PluginProcessor::setStateInformation(const void*, int) {}

juce::AudioProcessor* JUCE_CALLTYPE createPluginFilter()
{
    return new PluginProcessor();
}
```

```cpp
#include "PluginEditor.h"

PluginEditor::PluginEditor(PluginProcessor& p)

PluginEditor::~PluginEditor() {}

void PluginEditor::paint(juce::Graphics& g)
{
    g.fillAll(juce::Colours::darkgrey);
    g.setColour(juce::Colours::white);
    g.setFont(16.0f);
    g.drawText("Plugin Template", getLocalBounds(), juce::Justification::centred, true);
}

void PluginEditor::resized() {}
```
