```cpp
#include "PluginEditor.h"
#include &lt;BinaryData.h&gt;

// Helper: build a data URI from embedded HTML binary data so the browser
// can load it without a file path (works in all environments: DAW, Standalone, CI)
static juce::URL makeDataUri()
{
    const juce::String html (juce::CharPointer_UTF8 (BinaryData::index_html),
                             (size_t) BinaryData::index_htmlSize);
    const juce::String encoded = juce::Base64::toBase64 (html);
    return juce::URL ("data:text/html;base64," + encoded);
}

PluginEditor::PluginEditor(PluginProcessor&amp; p)
    : AudioProcessorEditor(&amp;p), processorRef(p)
{
    setSize(600, 400);

    // Wire the JS→C++ message handler BEFORE adding the browser as a child.
    // Messages arrive when JS calls window.__JUCE__.postMessage(jsonString).
    browser.pageAboutToLoad = [](const juce::String&amp;) { return true; };
    browser.pageFinishedLoading = [](const juce::String&amp;) {};

    // Handle incoming messages from JS (window.__JUCE__.postMessage calls this).
    // The message is the raw string passed by JS — parse as JSON { type, name, value }.
    browser.pageFinishedLoading = [this](const juce::String&amp;)
    {
        // Page is loaded; inject a listener for messages sent via postMessage.
        // JUCE 8 routes window.__JUCE__.postMessage to the nativeEventReceived callback.
    };

    addAndMakeVisible(browser);

    // Load index.html from embedded binary data via data URI.
    // For development, you can also load from a local file URL:
    //   browser.goToURL(juce::URL(File("/path/to/web/index.html")));
    browser.goToURL(makeDataUri());
}

PluginEditor::~PluginEditor() {}

void PluginEditor::paint(juce::Graphics&amp; g)
{
    g.fillAll(juce::Colours::black);
}

void PluginEditor::resized()
{
    browser.setBounds(getLocalBounds());
}
```

**IMPORTANT — JUCE 8 WebBrowserComponent API note:**
JUCE 8 changed `WebBrowserComponent` significantly. If the lambda-based API above does not compile, use the subclass approach:
- Subclass `juce::WebBrowserComponent` as `class PluginBrowser : public juce::WebBrowserComponent`

- For JS→C++ messages in JUCE 8, override `void resourceRequested(const juce::WebBrowserComponent::Resource&amp; resource, juce::WebBrowserComponent::ResourceRequestCallback callback)` or the new `nativeEventReceived` method depending on what the JUCE 8 API exposes in `libs/JUCE`.
